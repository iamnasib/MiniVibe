```js
// Wait for DOM to load before running scripts
document.addEventListener('DOMContentLoaded', () => {
  // NAVIGATION TOGGLE FOR MOBILE
  const navToggle = document.getElementById('nav-toggle');
  const navMenu = document.getElementById('primary-menu');

  navToggle.addEventListener('click', () => {
    const expanded = navToggle.getAttribute('aria-expanded') === 'true' || false;
    navToggle.setAttribute('aria-expanded', !expanded);
    navMenu.classList.toggle('active');
  });

  // COLLAPSE MOBILE MENU WHEN A LINK IS CLICKED
  navMenu.querySelectorAll('a').forEach(link => {
    link.addEventListener('click', () => {
      if (navMenu.classList.contains('active')) {
        navMenu.classList.remove('active');
        navToggle.setAttribute('aria-expanded', false);
      }
    });
  });

  // SMOOTH SCROLL FOR ALL INTERNAL LINKS
  const internalLinks = document.querySelectorAll('a[href^="#"]');
  internalLinks.forEach(link => {
    link.addEventListener('click', e => {
      const targetId = link.getAttribute('href').slice(1);
      const targetElement = document.getElementById(targetId);
      if (targetElement) {
        e.preventDefault();
        targetElement.focus({preventScroll:true});
        targetElement.scrollIntoView({behavior: 'smooth', block: 'start'});
        // For accessibility, focus outline visible after smooth scroll
      }
    });
  });

  // INTERACTIVE MENU ITEMS: KEYBOARD + MOUSE INTERACTION
  const menuItems = document.querySelectorAll('.menu-item');
  menuItems.forEach(item => {
    // On Enter or Space key, simulate click or maybe toggle details
    item.addEventListener('keydown', e => {
      if (e.key === 'Enter' || e.key === ' ') {
        e.preventDefault();
        item.click();
      }
    });

    // Optional: click effect, like a flash or animation (can be added here if needed)
  });

  // GALLERY SLIDER FUNCTIONALITY
  const galleryImagesWrapper = document.querySelector('.gallery-images');
  const galleryImages = galleryImagesWrapper.querySelectorAll('img');
  const prevBtn = document.querySelector('.gallery-prev');
  const nextBtn = document.querySelector('.gallery-next');
  let currentIndex = 0;

  // Set initial visible image
  function updateGallery() {
    // Scroll galleryImagesWrapper to the current image smoothly
    const imageWidth = galleryImages[0].clientWidth;
    galleryImagesWrapper.scrollTo({
      left: imageWidth * currentIndex,
      behavior: 'smooth',
    });
    // Update aria-live region by setting aria-atomic on galleryImagesWrapper (already done)
  }

  function showNextImage() {
    currentIndex = (currentIndex + 1) % galleryImages.length;
    updateGallery();
  }

  function showPrevImage() {
    currentIndex = (currentIndex - 1 + galleryImages.length) % galleryImages.length;
    updateGallery();
  }

  nextBtn.addEventListener('click', showNextImage);
  prevBtn.addEventListener('click', showPrevImage);

  // Keyboard support for gallery buttons
  [prevBtn, nextBtn].forEach(button => {
    button.addEventListener('keydown', e => {
      if (e.key === 'Enter' || e.key === ' ') {
        e.preventDefault();
        button.click();
      }
    });
  });

  // Swipe support on gallery for touch devices
  let startX = 0;
  let isDragging = false;

  galleryImagesWrapper.addEventListener('touchstart', e => {
    startX = e.touches[0].clientX;
    isDragging = true;
  });

  galleryImagesWrapper.addEventListener('touchmove', e => {
    if (!isDragging) return;
    let currentX = e.touches[0].clientX;
    let diffX = startX - currentX;
    // To avoid accidental swipe triggering while scrolling vertically, set threshold
    if (Math.abs(diffX) > 30) {
      if (diffX > 0) {
        showNextImage();
      } else {
        showPrevImage();
      }
      isDragging = false;
    }
  });

  galleryImagesWrapper.addEventListener('touchend', () => {
    isDragging = false;
  });

  // FORM VALIDATION AND SUBMISSION HANDLING
  const form = document.getElementById('contact-form');
  const formFeedback = document.getElementById('form-feedback');

  form.addEventListener('submit', e => {
    e.preventDefault();

    // Clear previous errors and feedback
    const errorMessages = form.querySelectorAll('.error-message');
    errorMessages.forEach(span => {
      span.textContent = '';
    });
    formFeedback.textContent = '';

    let valid = true;

    // Validate Name
    const nameInput = form.elements['name'];
    if (!nameInput.value.trim()) {
      showError(nameInput, 'Please enter your name.');
      valid = false;
    }

    // Validate Email
    const emailInput = form.elements['email'];
    if (!emailInput.value.trim()) {
      showError(emailInput, 'Please enter your email.');
      valid = false;
    } else if (!isValidEmail(emailInput.value.trim())) {
      showError(emailInput, 'Please enter a valid email address.');
      valid = false;
    }

    // Validate Message
    const messageInput = form.elements['message'];
    if (!messageInput.value.trim()) {
      showError(messageInput, 'Please enter a message.');
      valid = false;
    }

    if (!valid) {
      formFeedback.textContent = 'Please fix errors before submitting.';
      formFeedback.style.color = '#d0342c'; // red
      formFeedback.focus();
      return;
    }

    // Simulate form submission
    formFeedback.style.color = '#a94507'; // brand color brown/red
    formFeedback.textContent = 'Sending message...';

    // Simulate async delay (e.g., server request)
    setTimeout(() => {
      formFeedback.textContent = 'Thank you! Your message has been sent.';
      form.reset();
    }, 1500);
  });

  // Helper function to show error messages
  function showError(inputElement, message) {
    const errorSpan = inputElement.parentElement.querySelector('.error-message');
    if (errorSpan) {
      errorSpan.textContent = message;
    }
    inputElement.setAttribute('aria-invalid', 'true');
    inputElement.focus();
  }

  // Helper function to validate email with simple regex
  function isValidEmail(email) {
    // RFC 5322 simple email regex approximation
    const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return re.test(email.toLowerCase());
  }
});
```
