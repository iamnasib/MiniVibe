FILE: app.js
```js
// Select DOM elements
const todoForm = document.getElementById('todo-form');
const todoInput = document.getElementById('todo-input');
const todoList = document.getElementById('todo-list');
const todoCountEl = document.getElementById('todo-count');
const filterButtons = document.querySelectorAll('.filter-btn');
const clearCompletedBtn = document.getElementById('clear-completed');

let todos = [];
let currentFilter = 'all';

// Initialize app
document.addEventListener('DOMContentLoaded', () => {
  loadTodosFromStorage();
  renderTodos();
  updateTodoCount();
  updateFilterButtons();
});

/////////////////////////
// Storage
/////////////////////////

function saveTodosToStorage() {
  localStorage.setItem('todos', JSON.stringify(todos));
}

function loadTodosFromStorage() {
  const stored = localStorage.getItem('todos');
  if (stored) {
    try {
      todos = JSON.parse(stored);
      if (!Array.isArray(todos)) todos = [];
    } catch {
      todos = [];
    }
  }
}

/////////////////////////
// Render & Update UI
/////////////////////////

function renderTodos() {
  // Clear list
  todoList.innerHTML = '';

  // Filter todos according to currentFilter
  const filtered = todos.filter(todo => {
    if (currentFilter === 'all') return true;
    if (currentFilter === 'active') return !todo.completed;
    if (currentFilter === 'completed') return todo.completed;
    return true;
  });

  if (filtered.length === 0) {
    const emptyItem = document.createElement('li');
    emptyItem.className = 'todo-item empty';
    emptyItem.textContent = 'No todos to display.';
    emptyItem.style.textAlign = 'center';
    emptyItem.style.color = '#999';
    emptyItem.style.userSelect = 'none';
    todoList.appendChild(emptyItem);
    return;
  }

  // Create todo elements
  filtered.forEach(todo => {
    const li = document.createElement('li');
    li.className = 'todo-item';
    li.dataset.id = todo.id;
    if (todo.completed) li.classList.add('completed');

    // Checkbox
    const checkbox = document.createElement('input');
    checkbox.type = 'checkbox';
    checkbox.checked = todo.completed;
    checkbox.setAttribute('aria-label', 'Mark todo as completed');
    checkbox.tabIndex = 0;
    checkbox.addEventListener('change', () => toggleTodoCompleted(todo.id));

    // Text span
    const span = document.createElement('span');
    span.className = 'todo-text';
    span.textContent = todo.text;
    span.tabIndex = -1;
    span.setAttribute('aria-label', `Todo: ${todo.text}`);

    // Delete button
    const delBtn = document.createElement('button');
    delBtn.className = 'delete-btn';
    delBtn.setAttribute('aria-label', `Delete todo: ${todo.text}`);
    delBtn.innerHTML = '&times;';
    delBtn.addEventListener('click', () => deleteTodo(todo.id, li));

    li.appendChild(checkbox);
    li.appendChild(span);
    li.appendChild(delBtn);

    todoList.appendChild(li);
  });
  
  updateTodoCount();
}

function updateTodoCount() {
  const activeCount = todos.filter(t => !t.completed).length;
  todoCountEl.textContent = `${activeCount} item${activeCount !== 1 ? 's' : ''} left`;
}

function updateFilterButtons() {
  filterButtons.forEach(button => {
    const filter = button.dataset.filter;
    const isActive = filter === currentFilter;
    button.classList.toggle('active', isActive);
    button.setAttribute('aria-pressed', isActive.toString());
  });
}

/////////////////////////
// Event Handlers
/////////////////////////

todoForm.addEventListener('submit', event => {
  event.preventDefault();
  const text = todoInput.value.trim();
  if (text.length === 0) return;

  addTodo(text);
  todoInput.value = '';
  todoInput.focus();
});

filterButtons.forEach(button => {
  button.addEventListener('click', () => {
    const filter = button.dataset.filter;
    if (filter === currentFilter) return;
    currentFilter = filter;
    updateFilterButtons();
    renderTodos();
  });
});

clearCompletedBtn.addEventListener('click', () => {
  clearCompletedTodos();
});

/////////////////////////
// Todo Operations
/////////////////////////

function addTodo(text) {
  const newTodo = {
    id: generateId(),
    text,
    completed: false,
  };
  todos.push(newTodo);
  saveTodosToStorage();
  renderTodos();
}

function deleteTodo(id, element) {
  // Add removing animation class
  element.classList.add('removing');

  // After animation, remove from data and DOM
  element.addEventListener('transitionend', () => {
    todos = todos.filter(todo => todo.id !== id);
    saveTodosToStorage();
    renderTodos();
  }, { once: true });
}

function toggleTodoCompleted(id) {
  const todo = todos.find(t => t.id === id);
  if (!todo) return;

  todo.completed = !todo.completed;
  saveTodosToStorage();
  renderTodos();
}

function clearCompletedTodos() {
  const completedExists = todos.some(t => t.completed);
  if (!completedExists) return;

  // Animate the removal of all completed todo elements before clearing data and re-rendering
  const completedItems = Array.from(todoList.querySelectorAll('.todo-item.completed'));
  if (completedItems.length === 0) return;

  let animationsCompleted = 0;
  const onAnimEnd = () => {
    animationsCompleted++;
    if (animationsCompleted === completedItems.length) {
      todos = todos.filter(t => !t.completed);
      saveTodosToStorage();
      renderTodos();
    }
  };

  completedItems.forEach(item => {
    item.classList.add('removing');
    item.addEventListener('transitionend', onAnimEnd, { once: true });
  });
}

/////////////////////////
// Utils
/////////////////////////

function generateId() {
  // Simple unique ID generator based on timestamp and random number
  return Date.now().toString(36) + Math.random().toString(36).substr(2, 9);
}
```
